=== git pull ===
# 获取远程分支上的更新
# 会自动与本地分支合并
# 同一位置的不同修改会引发冲突
git pull

# 强制用远程分支覆盖本地分支
# 抛弃本地修改
git fetch -f
# 如果本地有尚未提交的修改，应先执行
git add -A
git commit -m "commit before fetch -f"

=== git push ===
git add -A
git commit -m "your message"
git push

=== collision ===
# 如果不同的人修改了不同的位置，不会冲突
# 如果不同的人修改了同一行，则会引起冲突

# 强制推送，覆盖远程分支
git push -u origin master -f 

# 先做pull,git会尝试把远程分支内容与本地分支合并
git pull

# 如果有冲突，查看冲突
git status
# 手工解决，修改冲突内容
git add -A
git commit -m "your message"
git push
